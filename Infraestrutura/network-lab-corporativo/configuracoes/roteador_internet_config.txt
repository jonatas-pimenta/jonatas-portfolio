! Configuração do Roteador Internet (Gateway WAN)
! Data: $(date)

! Configuração básica
hostname Router-Internet
enable secret cisco123

! Interface LAN para rede corporativa
interface GigabitEthernet0/0
 description LAN_Corporativa
 ip address 200.200.200.1 255.255.255.252
 no shutdown

! Interface WAN simulada (Internet)
interface GigabitEthernet0/1
 description Internet_Simulada
 ip address 8.8.8.1 255.255.255.252
 no shutdown

! Configuração de roteamento estático
ip route 192.168.0.0 255.255.0.0 200.200.200.2
ip route 0.0.0.0 0.0.0.0 8.8.8.2

! Pool DHCP para rede de gerência (opcional)
ip dhcp pool GERENCIA
 network 192.168.99.0 255.255.255.0
 default-router 192.168.99.1
 dns-server 8.8.8.8 8.8.4.4

! Configuração NAT (se necessário)
access-list 1 permit 200.200.200.0 0.0.0.3
ip nat inside source list 1 interface GigabitEthernet0/1 overload

! Definir interfaces NAT
interface GigabitEthernet0/0
 ip nat inside
interface GigabitEthernet0/1
 ip nat outside

! Configurações de segurança
line con 0
 password cisco123
 login
line vty 0 4
 password cisco123
 login

! Salvar configuração
end
copy running-config startup-config
