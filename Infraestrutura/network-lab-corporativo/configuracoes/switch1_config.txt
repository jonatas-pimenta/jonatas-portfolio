! Configuração do Switch1 - Switch da Filial
! Data: $(date)

! Configuração básica
hostname Switch-Filial
enable secret cisco123

! Configuração de VLANs
vlan 10
 name TI_Filial
vlan 20
 name Vendas_Filial
vlan 99
 name Gerencia_Filial

! Configuração de interfaces de acesso
interface range fa0/1-12
 switchport mode access
 switchport access vlan 10
 no shutdown

interface range fa0/13-24
 switchport mode access
 switchport access vlan 20
 no shutdown

! Interface trunk para conexão WAN/Roteador
interface gi0/1
 switchport mode trunk
 switchport trunk allowed vlan 10,20,99
 no shutdown

! Interface de gerência
interface vlan 99
 ip address 192.168.199.2 255.255.255.0
 no shutdown

! Gateway padrão
ip default-gateway 192.168.199.1

! Configuração de spanning-tree
spanning-tree mode pvst
spanning-tree vlan 1-99 priority 32768

! Configurações de segurança
line con 0
 password cisco123
 login
line vty 0 4
 password cisco123
 login

! Salvar configuração
end
copy running-config startup-config
